### 1-2. 구현 (GREEN)
`DxfParser.kt`에 `decodeDxfText()` 확장 함수 추가:
```kotlin
private fun decodeDxfText(raw: String): String = raw
    .replace(Regex("%%[Dd]"), "°")
    .replace(Regex("%%[Pp]"), "±")
    .replace(Regex("%%[Cc]"), "⌀")
    .replace(Regex("%%[Uu]"), "")   // 밑줄 토글 제거
    .replace(Regex("%%[Oo]"), "")   // 윗줄 토글 제거
    .replace(Regex("%%([0-9]{3})")) { mr ->  // %%nnn → ASCII 문자
        mr.groupValues[1].toIntOrNull()?.toChar()?.toString() ?: mr.value
    }
```

`parseText()` 마지막 줄: `return DxfText(layer, Vec2(x, y), height, decodeDxfText(text), rotation)`  
`parseMText()` 마지막 줄: `return DxfMText(layer, Vec2(x, y), height, decodeDxfText(textParts.joinToString("")), rotation)`

```kotlin
private fun trimmedBoundingBox(entities: List<DxfEntity>, trimRatio: Double = 0.05): BoundingBox? {
    val points = mutableListOf<Vec2>()
    // calculateBoundingBox()와 동일하게 points 수집
    // ...
    if (points.size < 4) return calculateBoundingBox(entities)
    val xs = points.map { it.x }.sorted()
    val ys = points.map { it.y }.sorted()
    val lo = (points.size * trimRatio).toInt().coerceAtLeast(0)
    val hi = (points.size * (1 - trimRatio)).toInt().coerceAtMost(points.size - 1)
    if (lo >= hi) return calculateBoundingBox(entities)
    return BoundingBox(xs[lo], ys[lo], xs[hi], ys[hi])
}
```

## Task 3: 뷰포트 컬링

**파일:** `DxfEntity.kt`, `EntityRenderer.kt`, `CoordTransform.kt`, `DrawingView.kt`

### 3-1. 엔티티 AABB 확장 함수
`DxfEntity.kt` 또는 새 파일 `EntityBounds.kt` 추가:
```kotlin
fun DxfEntity.worldBounds(): BoundingBox? = when (this) {
    is DxfLine       -> BoundingBox(minOf(start.x,end.x), minOf(start.y,end.y),
                                    maxOf(start.x,end.x), maxOf(start.y,end.y))
    is DxfCircle     -> BoundingBox(center.x-radius, center.y-radius,
                                    center.x+radius, center.y+radius)
    is DxfArc        -> BoundingBox(center.x-radius, center.y-radius,
                                    center.x+radius, center.y+radius) // 근사
    is DxfLwPolyline -> if (vertices.isEmpty()) null else
                        BoundingBox(vertices.minOf{it.x}, vertices.minOf{it.y},
                                    vertices.maxOf{it.x}, vertices.maxOf{it.y})
    is DxfText, is DxfMText -> null  // 텍스트는 항상 그린다 (크기 불명확)
    else -> null
}

fun BoundingBox.intersects(other: BoundingBox) =
    minX <= other.maxX && maxX >= other.minX && minY <= other.maxY && maxY >= other.minY
```

### 3-2. CoordTransform에 역변환 추가
```kotlin
fun screenToWorldBounds(screenW: Int, screenH: Int, matrix: Matrix): BoundingBox {
    val inv = Matrix(); matrix.invert(inv)
    val pts = floatArrayOf(0f, 0f, screenW.toFloat(), screenH.toFloat())
    inv.mapPoints(pts)
    return BoundingBox(minOf(pts[0],pts[2]).toDouble(), minOf(pts[1],pts[3]).toDouble(),
                       maxOf(pts[0],pts[2]).toDouble(), maxOf(pts[1],pts[3]).toDouble())
}
```

### 3-3. DrawingView.onDraw() 수정
```kotlin
override fun onDraw(canvas: Canvas) {

    renderer.drawAll(d.entities, canvas, matrix, viewport)
```

### 3-4. EntityRenderer.drawAll() 수정
```kotlin
fun drawAll(entities: List<DxfEntity>, canvas: Canvas, matrix: Matrix,
            viewport: BoundingBox? = null) {
    entities.forEach { entity ->
        if (entity !is DxfText && entity !is DxfMText) {
            val bounds = entity.worldBounds()
            if (bounds == null || viewport == null || bounds.intersects(viewport)) {
                draw(entity, canvas, matrix)
            }
        }
    }
    // 텍스트는 뷰포트 무관하게 그린다 (크기 불명확)
    entities.forEach { if (it is DxfText || it is DxfMText) draw(it, canvas, matrix) }
}
```
